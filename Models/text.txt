Table Customers {
  Id int [primary key]
  Name varchar
  LastName varchar
  Telephone char(15)
  AFM int
  Type Enum('Retail', 'Wholesale')
}

Table Orders {
  Id int [primary key]
  CostumerId int
  Delivered bool
  SaleChannel varchar
  OrderComment varchar
}

Table Products {
  Id int [primary key]
  OrderId int
  CategoryId int
  ModelId int
  IsComplete bool
  DateOfCreation date
  ExpectedStartDate date
  ExpectedFinishDate date
  ProductComment varchar
}

Table ProductCategories {
  Id int [primary key]
  CategoryName varchar
}

Table Models {
  Id int [primary key]
  ModelName varchar
}

Table Tasks {
  Id int [primary key]
  ProcessId int
  ProductId int
  IsComplete bool
}

Table Processes {
  Id int [primary key]
  Name varchar
}

Table Users {
  Id int [primary key]
  UserName varchar
  ImageURL varchar
}

Table UserProcesses {
  Id int [primary key]
  UserId int
  ProcessId int
}

Table TaskDependencies {
  Id int [primary key]
  TaskId int
  DependsOnId int
}

Table TaskAttributes {
  Id int [primary key]
  TaskId int
  ValueId int
}

Table AttributeValues {
  Id int [primary key]
  AttributeName varchar
  AttributeValue varchar
}

Table Patterns {
  Id int [primary key]
  ModelId int
  PatternName varchar
}

Table PatternAreas {
  Id int [primary key]
  PatternId int
  AreaName varchar
}

Table Fabrics {
  Id int [primary key]
  FabricName varchar
}

Table ThreadColors {
  Id int [primary key]
  ColorName varchar
}

Table SewingTypes {
  Id int [primary key]
  TypeName varchar
}

Table FabricParaeters {
  Id int [primary key]
  FabricParameterName varchar
}
